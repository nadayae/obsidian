## 📋 할 일
```dataviewjs
const ROSEWATER = "#e6aba9";
const projectName = dv.current().file.name;
const TASKS_PATH = '"05. Tasks"';

// 1. 데이터 수집
const allTasks = dv.pages(TASKS_PATH).filter(t => {
    const prop = t.프로젝트;
    if (!prop) return false;
    const nameOnly = prop.path ? prop.fileName : String(prop).replace(/[\[\]]/g, "").trim();
    return nameOnly === projectName;
}).array();

// 2. 메인 레이아웃 구성
const container = dv.el("div", "");
container.style.display = "grid";
container.style.gridTemplateColumns = "320px 1fr";
container.style.gap = "25px";

// [좌측: 할 일 목록]
const leftPanel = container.createDiv({ style: "border-right: 1px solid #eee; padding-right: 15px;" });
leftPanel.innerHTML = `<h3 style="color:${ROSEWATER}; font-size:1rem; margin-bottom:15px;">미배정 할 일 📋</h3>`;
const listArea = leftPanel.createDiv({ style: "display:grid; gap:10px; max-height:550px; overflow-y:auto;" });

// [우측: 탭형 캘린더]
const rightPanel = container.createDiv();
rightPanel.innerHTML = `
    <div style="display:flex; justify-content:flex-end; gap:10px; margin-bottom:15px;">
        <button id="view-week" style="font-size:0.7rem; padding:4px 12px; border-radius:15px; border:1px solid ${ROSEWATER}; background:${ROSEWATER}1A; color:${ROSEWATER}; cursor:pointer;">Weekly</button>
        <button id="view-month" style="font-size:0.7rem; padding:4px 12px; border-radius:15px; border:1px solid #eee; background:none; color:#888; cursor:pointer;">Monthly</button>
    </div>
    <div id="cal-render-area" style="background:rgba(245,234,224,0.3); border:2px solid ${ROSEWATER}4D; border-radius:15px; min-height:500px; padding:20px;">
        </div>
`;

// 3. 달력 그리기 로직 (직접 구현)
const renderCalendar = (mode) => {
    const calArea = rightPanel.querySelector("#cal-render-area");
    let calHTML = `<div style="display:grid; grid-template-columns:repeat(7, 1fr); gap:10px; text-align:center;">`;
    
    const days = ['일','월','화','수','목','금','토'];
    days.forEach(d => calHTML += `<div style="font-size:0.75rem; font-weight:800; color:#8a81a3;">${d}</div>`);
    
    // 임시로 이번 주/달 날짜 생성 (드래그 타겟)
    for(let i=1; i<=31; i++) {
        calHTML += `
            <div class="cal-cell" data-date="${i}" style="height:80px; background:white; border-radius:8px; border:1px solid #eee; padding:5px; font-size:0.7rem; color:#aaa; cursor:pointer;">
                ${i}
                <div class="cell-task-area" style="margin-top:5px;"></div>
            </div>`;
    }
    calHTML += `</div>`;
    calArea.innerHTML = calHTML;
    
    // [드래그 앤 드롭 이벤트 바인딩]
    const cells = calArea.querySelectorAll(".cal-cell");
    cells.forEach(cell => {
        cell.ondragover = (e) => e.preventDefault();
        cell.ondrop = (e) => {
            e.preventDefault();
            const taskName = e.dataTransfer.getData("taskName");
            cell.querySelector(".cell-task-area").innerHTML += `
                <div style="background:${ROSEWATER}33; color:${ROSEWATER}; padding:2px 4px; border-radius:4px; font-size:0.6rem; margin-bottom:2px;">
                    ${taskName}
                </div>`;
            alert(`[보안 알림] ${taskName} 일정이 ${cell.dataset.date}일로 가배정되었습니다. (실제 파일 수정은 수동으로 해주세요!)`);
        };
    });
};

// 4. 할 일 리스트 렌더링
const renderList = () => {
    const unscheduled = allTasks.filter(t => !t.날짜 && t.완료여부 !== true);
    listArea.innerHTML = unscheduled.map(t => {
        const div = document.createElement("div");
        div.draggable = true;
        div.style = `background:var(--background-primary); border:1.5px solid #eee; border-radius:10px; padding:12px; cursor:grab; font-size:0.8rem; font-weight:700;`;
        div.innerHTML = `<a class="internal-link" href="${t.file.path}">${t.file.name}</a>`;
        div.ondragstart = (e) => e.dataTransfer.setData("taskName", t.file.name);
        return div.outerHTML;
    }).join("");
};

renderCalendar("weekly");
renderList();
```

## ✅ 완료된 할 일

```dataview
TABLE 구분 AS "구분", 날짜 AS "날짜"

SORT file.mtime DESC
LIMIT 5
```
